    Builds the sample scheme with settings for :critical, :error, :warning, :notice, :info, :debug, :row_even, and :row_odd colors.

def initialize( h = nil )
  scheme = {
    :critical => [ :yellow, :on_red  ],
    :error    => [ :bold,   :red     ],
    :warning  => [ :bold,   :yellow  ],
    :notice   => [ :bold,   :magenta ],
    :info     => [ :bold,   :cyan    ],
    :debug    => [ :bold,   :green   ],
    :row_even => [ :cyan    ],
    :row_odd  => [ :magenta ]
  }
  super(scheme)
end
